# 神经网络核心概念

## 1. 前向传播（Forward Propagation）

### 1.1 基本原理

前向传播是神经网络从输入到输出的计算过程。

```python
import numpy as np

# 简单的前向传播示例
def forward_pass(X, W1, b1, W2, b2):
    """
    X: 输入数据 (batch_size, input_dim)
    W1: 第一层权重 (input_dim, hidden_dim)
    b1: 第一层偏置 (hidden_dim,)
    W2: 第二层权重 (hidden_dim, output_dim)
    b2: 第二层偏置 (output_dim,)
    """
    # 第一层
    z1 = np.dot(X, W1) + b1      # 线性变换
    a1 = np.maximum(0, z1)        # ReLU激活
    
    # 第二层
    z2 = np.dot(a1, W2) + b2     # 线性变换
    a2 = softmax(z2)              # Softmax激活
    
    return a2, (z1, a1, z2)       # 返回输出和中间值

def softmax(x):
    exp_x = np.exp(x - np.max(x, axis=1, keepdims=True))
    return exp_x / np.sum(exp_x, axis=1, keepdims=True)
```

### 1.2 多层网络前向传播

```python
class NeuralNetwork:
    def __init__(self, layer_sizes):
        """
        layer_sizes: [input_dim, hidden1, hidden2, ..., output_dim]
        """
        self.weights = []
        self.biases = []
        
        for i in range(len(layer_sizes) - 1):
            W = np.random.randn(layer_sizes[i], layer_sizes[i+1]) * 0.01
            b = np.zeros(layer_sizes[i+1])
            self.weights.append(W)
            self.biases.append(b)
    
    def forward(self, X):
        """前向传播"""
        self.activations = [X]
        self.z_values = []
        
        for i, (W, b) in enumerate(zip(self.weights, self.biases)):
            z = np.dot(self.activations[-1], W) + b
            self.z_values.append(z)
            
            # 最后一层用softmax，其他层用ReLU
            if i == len(self.weights) - 1:
                a = softmax(z)
            else:
                a = np.maximum(0, z)
            
            self.activations.append(a)
        
        return self.activations[-1]
```

## 2. 反向传播（Backpropagation）

### 2.1 基本原理

反向传播通过链式法则计算损失函数对每个参数的梯度。

```python
def backward_pass(X, y, cache, W1, W2):
    """
    X: 输入 (batch_size, input_dim)
    y: 标签 (batch_size,)
    cache: 前向传播的中间值 (z1, a1, z2)
    """
    z1, a1, z2 = cache
    m = X.shape[0]
    
    # 输出层梯度
    dz2 = softmax(z2)
    dz2[range(m), y] -= 1  # 交叉熵损失的梯度
    dz2 /= m
    
    # 第二层参数梯度
    dW2 = np.dot(a1.T, dz2)
    db2 = np.sum(dz2, axis=0)
    
    # 隐藏层梯度
    da1 = np.dot(dz2, W2.T)
    dz1 = da1 * (z1 > 0)  # ReLU的梯度
    
    # 第一层参数梯度
    dW1 = np.dot(X.T, dz1)
    db1 = np.sum(dz1, axis=0)
    
    return dW1, db1, dW2, db2
```

### 2.2 完整的反向传播实现

```python
class NeuralNetwork:
    # ... (前面的代码)
    
    def backward(self, y):
        """反向传播"""
        m = y.shape[0]
        grads = {'W': [], 'b': []}
        
        # 输出层梯度
        dz = self.activations[-1].copy()
        dz[range(m), y] -= 1
        dz /= m
        
        # 从后向前计算梯度
        for i in reversed(range(len(self.weights))):
            # 参数梯度
            dW = np.dot(self.activations[i].T, dz)
            db = np.sum(dz, axis=0)
            grads['W'].insert(0, dW)
            grads['b'].insert(0, db)
            
            if i > 0:
                # 传播到前一层
                da = np.dot(dz, self.weights[i].T)
                # ReLU梯度
                dz = da * (self.z_values[i-1] > 0)
        
        return grads
```

## 3. 损失函数（Loss Functions）

### 3.1 交叉熵损失（Cross-Entropy Loss）

用于分类任务。

```python
def cross_entropy_loss(y_pred, y_true):
    """
    y_pred: 预测概率 (batch_size, num_classes)
    y_true: 真实标签 (batch_size,)
    """
    m = y_true.shape[0]
    log_probs = -np.log(y_pred[range(m), y_true])
    loss = np.sum(log_probs) / m
    return loss

# 二分类交叉熵
def binary_cross_entropy(y_pred, y_true):
    """
    y_pred: 预测概率 (batch_size,)
    y_true: 真实标签 (batch_size,)
    """
    epsilon = 1e-15
    y_pred = np.clip(y_pred, epsilon, 1 - epsilon)
    loss = -np.mean(y_true * np.log(y_pred) + (1 - y_true) * np.log(1 - y_pred))
    return loss
```

### 3.2 均方误差损失（MSE Loss）

用于回归任务。

```python
def mse_loss(y_pred, y_true):
    """
    y_pred: 预测值 (batch_size, output_dim)
    y_true: 真实值 (batch_size, output_dim)
    """
    return np.mean((y_pred - y_true) ** 2)

def mse_loss_gradient(y_pred, y_true):
    """MSE损失的梯度"""
    return 2 * (y_pred - y_true) / y_true.shape[0]
```

### 3.3 Huber Loss

对异常值更鲁棒的损失函数。

```python
def huber_loss(y_pred, y_true, delta=1.0):
    """
    delta: 阈值参数
    """
    error = y_pred - y_true
    is_small_error = np.abs(error) <= delta
    
    squared_loss = 0.5 * error ** 2
    linear_loss = delta * (np.abs(error) - 0.5 * delta)
    
    return np.mean(np.where(is_small_error, squared_loss, linear_loss))
```

## 4. 激活函数（Activation Functions）

### 4.1 ReLU（Rectified Linear Unit）

```python
def relu(x):
    """ReLU激活函数"""
    return np.maximum(0, x)

def relu_derivative(x):
    """ReLU的导数"""
    return (x > 0).astype(float)
```

### 4.2 Sigmoid

```python
def sigmoid(x):
    """Sigmoid激活函数"""
    return 1 / (1 + np.exp(-x))

def sigmoid_derivative(x):
    """Sigmoid的导数"""
    s = sigmoid(x)
    return s * (1 - s)
```

### 4.3 Tanh

```python
def tanh(x):
    """Tanh激活函数"""
    return np.tanh(x)

def tanh_derivative(x):
    """Tanh的导数"""
    return 1 - np.tanh(x) ** 2
```

### 4.4 Softmax

```python
def softmax(x):
    """Softmax激活函数"""
    exp_x = np.exp(x - np.max(x, axis=-1, keepdims=True))
    return exp_x / np.sum(exp_x, axis=-1, keepdims=True)
```

### 4.5 GELU（Gaussian Error Linear Unit）

大模型中常用的激活函数。

```python
def gelu(x):
    """GELU激活函数"""
    return 0.5 * x * (1 + np.tanh(np.sqrt(2 / np.pi) * (x + 0.044715 * x**3)))

# 或使用更精确的版本
from scipy.special import erf

def gelu_exact(x):
    return 0.5 * x * (1 + erf(x / np.sqrt(2)))
```

## 5. 梯度下降（Gradient Descent）

### 5.1 批量梯度下降（Batch GD）

```python
def batch_gradient_descent(X, y, learning_rate=0.01, epochs=1000):
    """批量梯度下降"""
    m, n = X.shape
    W = np.zeros(n)
    b = 0
    
    for epoch in range(epochs):
        # 前向传播
        y_pred = np.dot(X, W) + b
        
        # 计算梯度
        dW = (1/m) * np.dot(X.T, (y_pred - y))
        db = (1/m) * np.sum(y_pred - y)
        
        # 更新参数
        W -= learning_rate * dW
        b -= learning_rate * db
        
        if epoch % 100 == 0:
            loss = mse_loss(y_pred, y)
            print(f"Epoch {epoch}, Loss: {loss:.4f}")
    
    return W, b
```

### 5.2 随机梯度下降（SGD）

```python
def stochastic_gradient_descent(X, y, learning_rate=0.01, epochs=1000):
    """随机梯度下降"""
    m, n = X.shape
    W = np.zeros(n)
    b = 0
    
    for epoch in range(epochs):
        # 随机打乱数据
        indices = np.random.permutation(m)
        
        for i in indices:
            # 单个样本的前向传播
            xi = X[i:i+1]
            yi = y[i:i+1]
            y_pred = np.dot(xi, W) + b
            
            # 计算梯度
            dW = np.dot(xi.T, (y_pred - yi))
            db = np.sum(y_pred - yi)
            
            # 更新参数
            W -= learning_rate * dW
            b -= learning_rate * db
    
    return W, b
```

### 5.3 小批量梯度下降（Mini-batch GD）

```python
def mini_batch_gradient_descent(X, y, batch_size=32, learning_rate=0.01, epochs=100):
    """小批量梯度下降"""
    m, n = X.shape
    W = np.zeros(n)
    b = 0
    
    for epoch in range(epochs):
        # 随机打乱数据
        indices = np.random.permutation(m)
        X_shuffled = X[indices]
        y_shuffled = y[indices]
        
        # 分批处理
        for i in range(0, m, batch_size):
            X_batch = X_shuffled[i:i+batch_size]
            y_batch = y_shuffled[i:i+batch_size]
            
            # 前向传播
            y_pred = np.dot(X_batch, W) + b
            
            # 计算梯度
            batch_m = X_batch.shape[0]
            dW = (1/batch_m) * np.dot(X_batch.T, (y_pred - y_batch))
            db = (1/batch_m) * np.sum(y_pred - y_batch)
            
            # 更新参数
            W -= learning_rate * dW
            b -= learning_rate * db
    
    return W, b
```

## 6. 优化器（Optimizers）

### 6.1 Momentum

```python
class MomentumOptimizer:
    def __init__(self, learning_rate=0.01, momentum=0.9):
        self.lr = learning_rate
        self.momentum = momentum
        self.velocity = {}
    
    def update(self, params, grads):
        if not self.velocity:
            for key in params:
                self.velocity[key] = np.zeros_like(params[key])
        
        for key in params:
            self.velocity[key] = self.momentum * self.velocity[key] - self.lr * grads[key]
            params[key] += self.velocity[key]
```

### 6.2 Adam

```python
class AdamOptimizer:
    def __init__(self, learning_rate=0.001, beta1=0.9, beta2=0.999, epsilon=1e-8):
        self.lr = learning_rate
        self.beta1 = beta1
        self.beta2 = beta2
        self.epsilon = epsilon
        self.m = {}
        self.v = {}
        self.t = 0
    
    def update(self, params, grads):
        if not self.m:
            for key in params:
                self.m[key] = np.zeros_like(params[key])
                self.v[key] = np.zeros_like(params[key])
        
        self.t += 1
        
        for key in params:
            # 更新一阶矩估计
            self.m[key] = self.beta1 * self.m[key] + (1 - self.beta1) * grads[key]
            # 更新二阶矩估计
            self.v[key] = self.beta2 * self.v[key] + (1 - self.beta2) * (grads[key] ** 2)
            
            # 偏差修正
            m_hat = self.m[key] / (1 - self.beta1 ** self.t)
            v_hat = self.v[key] / (1 - self.beta2 ** self.t)
            
            # 更新参数
            params[key] -= self.lr * m_hat / (np.sqrt(v_hat) + self.epsilon)
```

## 7. 完整训练示例

```python
class SimpleNN:
    def __init__(self, input_dim, hidden_dim, output_dim, learning_rate=0.01):
        # 初始化参数
        self.W1 = np.random.randn(input_dim, hidden_dim) * 0.01
        self.b1 = np.zeros(hidden_dim)
        self.W2 = np.random.randn(hidden_dim, output_dim) * 0.01
        self.b2 = np.zeros(output_dim)
        self.lr = learning_rate
    
    def forward(self, X):
        """前向传播"""
        self.z1 = np.dot(X, self.W1) + self.b1
        self.a1 = np.maximum(0, self.z1)  # ReLU
        self.z2 = np.dot(self.a1, self.W2) + self.b2
        self.a2 = softmax(self.z2)
        return self.a2
    
    def backward(self, X, y):
        """反向传播"""
        m = X.shape[0]
        
        # 输出层梯度
        dz2 = self.a2.copy()
        dz2[range(m), y] -= 1
        dz2 /= m
        
        # 第二层参数梯度
        dW2 = np.dot(self.a1.T, dz2)
        db2 = np.sum(dz2, axis=0)
        
        # 隐藏层梯度
        da1 = np.dot(dz2, self.W2.T)
        dz1 = da1 * (self.z1 > 0)
        
        # 第一层参数梯度
        dW1 = np.dot(X.T, dz1)
        db1 = np.sum(dz1, axis=0)
        
        # 更新参数
        self.W1 -= self.lr * dW1
        self.b1 -= self.lr * db1
        self.W2 -= self.lr * dW2
        self.b2 -= self.lr * db2
    
    def train(self, X_train, y_train, X_val, y_val, epochs=100, batch_size=32):
        """训练模型"""
        m = X_train.shape[0]
        
        for epoch in range(epochs):
            # 随机打乱数据
            indices = np.random.permutation(m)
            X_shuffled = X_train[indices]
            y_shuffled = y_train[indices]
            
            # 小批量训练
            for i in range(0, m, batch_size):
                X_batch = X_shuffled[i:i+batch_size]
                y_batch = y_shuffled[i:i+batch_size]
                
                # 前向传播和反向传播
                self.forward(X_batch)
                self.backward(X_batch, y_batch)
            
            # 验证
            if epoch % 10 == 0:
                train_loss = self.compute_loss(X_train, y_train)
                val_loss = self.compute_loss(X_val, y_val)
                train_acc = self.compute_accuracy(X_train, y_train)
                val_acc = self.compute_accuracy(X_val, y_val)
                
                print(f"Epoch {epoch}")
                print(f"  Train Loss: {train_loss:.4f}, Acc: {train_acc:.4f}")
                print(f"  Val Loss: {val_loss:.4f}, Acc: {val_acc:.4f}")
    
    def compute_loss(self, X, y):
        """计算损失"""
        y_pred = self.forward(X)
        return cross_entropy_loss(y_pred, y)
    
    def compute_accuracy(self, X, y):
        """计算准确率"""
        y_pred = self.forward(X)
        predictions = np.argmax(y_pred, axis=1)
        return np.mean(predictions == y)
    
    def predict(self, X):
        """预测"""
        y_pred = self.forward(X)
        return np.argmax(y_pred, axis=1)
```
