y = np.array([i * 10 + np.random.randint(100) for i in x])  
y = np.array(y) + 1000  
print(x[:5])  
print(y[:5])  
  
x = x / 100  
y = y / 100  
  
epoch = 200  
w = 0.1  
b = 0.1  
  
for _ in range(epoch):  
    for i in range(len(x)):  
        y_predict = w * x[i] + b  
        loss = y_predict - y[i]  
        w = w - loss * x[i]  
        b = b - loss  
  
y_predict = w * x + b  
plt.scatter(x, y)  
plt.plot(x, y_predict, color='r')  
plt.show()

# 定义优化器：使用随机梯度下降（SGD），学习率为 0.01
# model.parameters() 告诉优化器需要更新哪些参数（权重和偏置）
optimizer = torch.optim.SGD(model.parameters(), lr=0.01)

# 定义损失函数：交叉熵损失，适用于多分类任务
criterion = nn.CrossEntropyLoss()

# 注意：range(1) 表示只训练 1 轮，实际使用时应改为更大的值，如 range(10)
num_epochs = 10
for epoch in range(num_epochs):
    for i, (inputs, labels) in enumerate(train_loader):
        # 将数据移动到指定设备（GPU 或 CPU）
        inputs = inputs.to(device)
        labels = labels.to(device)

        # 将输入数据展平为一维向量
        # 例如 28x28 的图片 -> 784 个数字
        inputs = inputs.view(inputs.shape[0], -1)

        # 第一步：前向传播，得到模型预测结果
        outputs = model(inputs)

        # 第二步：计算损失（预测值与真实标签的差距）
        loss = criterion(outputs, labels.long())

        # 第三步：清空上一轮的梯度（必须在反向传播前执行）
        optimizer.zero_grad()

        # 第四步：反向传播，计算各参数的梯度
        loss.backward()

        # 第五步：根据梯度更新模型参数
        optimizer.step()

        # 每 10 个批次打印一次训练进度
        if (i + 1) % 10 == 0:
            print('Epoch [{}/{}], Step [{}/{}], Loss: {:.4f}'
                  .format(epoch + 1, num_epochs, i + 1, len(train_loader), loss.item()))
